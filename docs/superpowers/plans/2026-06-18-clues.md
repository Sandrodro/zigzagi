# Clues Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Generate NYT-Monday-register Georgian clues for a filled puzzle's entries with
batched Gemini Pro calls, let the admin accept/edit/reject (regenerate) each clue with a full
audit log, and block publishing until every clue is accepted.

**Architecture:** Extends the `GeminiClient` (from the Sourcing & Pool plan) with a batched
`clue()` method using the **Pro** model (DESIGN.md §4.7). Each `Entry` already carries `clue`
+ `clue_status` (from the Walking Skeleton). A new `ClueEvent` table records every
accept/edit/reject for the KPI #3 accept-rate metric (DESIGN.md §4.1, §9, §10). A pure
publish-gate (`can_publish`) enforces "all accepted" (DESIGN.md §4.2.7).

**Tech Stack:** Builds on Walking Skeleton + Solver (entries exist) + Sourcing (AI client).

## Global Constraints

- **Clue model is config-driven**: `GEMINI_CLUE_MODEL` (default `gemini-2.5-pro`). Structured
  JSON output, **one bounded retry**, then surface an error — **never auto-publish on AI failure** (DESIGN.md §4.7, §10).
- **Batched** clue generation; cache by `(answer, theme)` is a later optimization (DESIGN.md §5) — not in this plan (YAGNI).
- `Entry.clue_status` ∈ `pending | generated | accepted | edited | rejected` (DESIGN.md §4.5).
- **Publish is blocked** until all of a puzzle's entries are `accepted` or `edited` (an edited
  clue is admin-authored, hence publishable) (DESIGN.md §4.2.7).
- Every review action is **logged** as a `ClueEvent` (audit + accept-rate metric).

import uuid

from sqlalchemy.orm import Session

from app.ai.client import ClueRequest, GeminiClient
from app.models import ClueEvent, Entry, Puzzle


def generate_clues(db: Session, puzzle: Puzzle, ai: GeminiClient) -> int:
    pending = [e for e in puzzle.entries if e.clue_status in ("pending", "rejected")]
    if not pending:
        return 0
    batch = [
        ClueRequest(
            entry_id=str(e.id), answer=e.answer, direction=e.direction,
            number=e.number, theme=puzzle.theme, source_snippet=None,
        )
        for e in pending
    ]
    results = {r.entry_id: r.clue for r in ai.clue(batch)}
    by_id = {str(e.id): e for e in pending}
    n = 0
    for eid, clue in results.items():
        entry = by_id.get(eid)
        if entry is not None:
            entry.clue = clue
            entry.clue_status = "generated"
            n += 1
    db.flush()
    return n


def review_clue(db: Session, entry_id: uuid.UUID, action: str, new_clue: str | None = None, ai: GeminiClient | None = None) -> Entry:
    entry = db.get(Entry, entry_id)
    old = entry.clue
    if action == "accept":
        entry.clue_status = "accepted"
        db.add(ClueEvent(id=uuid.uuid4(), entry_id=entry.id, action="accept", old_clue=old, new_clue=old))
    elif action == "edit":
        entry.clue = new_clue
        entry.clue_status = "edited"
        db.add(ClueEvent(id=uuid.uuid4(), entry_id=entry.id, action="edit", old_clue=old, new_clue=new_clue))
    elif action == "reject":
        db.add(ClueEvent(id=uuid.uuid4(), entry_id=entry.id, action="reject", old_clue=old, new_clue=None))
        entry.clue_status = "rejected"
        if ai is not None:
            puzzle = db.get(Puzzle, entry.puzzle_id)
            generate_clues(db, puzzle, ai)  # regenerates this rejected entry
    else:
        raise ValueError(f"unknown action {action}")
    db.flush()
    return entry


def accept_rate(db: Session, puzzle: Puzzle) -> float:
    reviewed = [e for e in puzzle.entries if e.clue_status in ("accepted", "edited", "rejected")]
    if not reviewed:
        return 0.0
    good = [e for e in reviewed if e.clue_status in ("accepted", "edited")]
    return len(good) / len(reviewed)
```

def can_publish(puzzle) -> tuple[bool, str | None]:
    unfinished = [e for e in puzzle.entries if e.clue_status not in ("accepted", "edited")]
    if not puzzle.entries:
        return False, "puzzle has no entries"
    if unfinished:
        return False, f"{len(unfinished)} clues not yet accepted"
    return True, None

from app.services.clues import generate_clues, review_clue


class ClueReviewRequest(BaseModel):
    action: str
    clue: str | None = None


@router.post("/puzzles/{puzzle_id}/clues")
def gen_clues(puzzle_id: uuid.UUID, db: Session = Depends(get_db), ai: GeminiClient = Depends(get_gemini)):
    puzzle = db.get(Puzzle, puzzle_id)
    if puzzle is None:
        raise HTTPException(404, "puzzle not found")
    n = generate_clues(db, puzzle, ai)
    db.commit()
    return {"generated": n}


@router.patch("/puzzles/{puzzle_id}/clues/{entry_id}")
def review(puzzle_id: uuid.UUID, entry_id: uuid.UUID, body: ClueReviewRequest, db: Session = Depends(get_db), ai: GeminiClient = Depends(get_gemini)):
    entry = review_clue(db, entry_id, body.action, body.clue, ai=ai)
    db.commit()
    return {"clue_status": entry.clue_status}
```
